  <script>
    // Simple script to add more follow-up items
    document.getElementById('addFollowup').addEventListener('click', function() {
      const container = document.getElementById('followupContainer');
      const newItem = document.querySelector('.followup-item').cloneNode(true);
      newItem.querySelector('input[type="date"]').value = '';
      newItem.querySelector('textarea').value = '';
      container.appendChild(newItem);
      
      // Add event listener to new remove button
      newItem.querySelector('.remove-followup').addEventListener('click', function() {
        if (document.querySelectorAll('.followup-item').length > 1) {
          this.closest('.followup-item').remove();
        } else {
          alert('At least one follow-up item is required.');
        }
      });
    });

    // Add event listener to initial remove button
    document.querySelector('.remove-followup').addEventListener('click', function() {
      if (document.querySelectorAll('.followup-item').length > 1) {
        this.closest('.followup-item').remove();
      } else {
        alert('At least one follow-up item is required.');
      }
    });
  </script>
